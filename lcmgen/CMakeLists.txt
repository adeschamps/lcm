macro(strip_flag FLAG VAR)
  string(REGEX REPLACE "${FLAG}" "" ${VAR} "${${VAR}}")
endmacro()
strip_flag("-fsanitize=(address|memory)" CMAKE_C_FLAGS)
strip_flag("-fsanitize=(address|memory)" CMAKE_EXE_LINKER_FLAGS)
